### 실험 A: CPU 멀티코어 대역폭
**목적**: CPU가 GPU prefetch helper로 사용 가능한지 확인

```python
# scripts/cache_experiments/260524_cpu_multicore_bw.py

import threading, time, numpy as np, os

def measure_multicore_bw(n_cores: int, array_size_mb: int = 256) -> float:
    """n개 코어가 동시에 memcpy 할 때의 합산 대역폭"""
    size = array_size_mb * 1024 * 1024 // 4
    results = []
    
    def worker():
        src = np.ones(size, dtype=np.float32)
        dst = np.empty_like(src)
        # 워밍업
        for _ in range(3): np.copyto(dst, src)
        
        t0 = time.perf_counter()
        for _ in range(10): np.copyto(dst, src)
        t1 = time.perf_counter()
        
        bw = 2 * size * 4 * 10 / (t1 - t0) / 1e9
        results.append(bw)
    
    threads = [threading.Thread(target=worker) for _ in range(n_cores)]
    for t in threads: t.start()
    for t in threads: t.join()
    
    return sum(results)

# 측정: 1, 2, 4, 7, 14 코어
for n in [1, 2, 4, 7, 14]:
    bw = measure_multicore_bw(n)
    print(f"{n:2d} cores: {bw:.1f} GB/s")
```

**예상 결과**:
- 1 core: ~33 GB/s
- 4 cores: ~100-130 GB/s
- 14 cores: ~150-220 GB/s (DRAM 포화로 수렴)

### 실험 B: CPU + GPU 동시 DRAM 대역폭
**목적**: Async pipeline에서 CPU prefetch + GPU compute 동시 실행 시 간섭 측정

```python
# scripts/cache_experiments/260524_cpu_gpu_concurrent_bw.py

import threading, torch, numpy as np

def measure_concurrent_bw(n_cpu_cores: int):
    """GPU 추론 중 CPU가 동시에 DRAM 접근할 때의 GPU 대역폭"""
    
    # CPU 워커 (계속 DRAM 접근)
    stop_flag = threading.Event()
    cpu_bw_results = []
    
    def cpu_worker():
        size = 256 * 1024 * 1024 // 4
        src = np.ones(size, dtype=np.float32)
        dst = np.empty_like(src)
        while not stop_flag.is_set():
            np.copyto(dst, src)
    
    # CPU 스레드 시작
    cpu_threads = [threading.Thread(target=cpu_worker) for _ in range(n_cpu_cores)]
    for t in cpu_threads: t.start()
    
    # GPU 대역폭 측정 (CPU 간섭 중)
    gpu_tensor = torch.zeros(256 * 1024 * 1024 // 4, dtype=torch.float32, device='cuda')
    out = torch.empty_like(gpu_tensor)
    
    # CUDA Graph로 측정
    g = torch.cuda.CUDAGraph()
    with torch.cuda.graph(g):
        for _ in range(100): out.copy_(gpu_tensor)
    
    times = []
    for _ in range(10):
        start = torch.cuda.Event(enable_timing=True)
        end   = torch.cuda.Event(enable_timing=True)
        start.record(); g.replay(); end.record()
        torch.cuda.synchronize()
        times.append(start.elapsed_time(end) / 100)
    
    stop_flag.set()
    for t in cpu_threads: t.join()
    
    mean_ms = np.median(times)
    size_bytes = gpu_tensor.numel() * 4
    bw = 2 * size_bytes / (mean_ms * 1e-3) / 1e9
    return bw

print(f"GPU alone:          {measure_concurrent_bw(0):.1f} GB/s")
print(f"GPU + 1 CPU core:   {measure_concurrent_bw(1):.1f} GB/s")
print(f"GPU + 4 CPU cores:  {measure_concurrent_bw(4):.1f} GB/s")
print(f"GPU + 14 CPU cores: {measure_concurrent_bw(14):.1f} GB/s")
```

## 3. Phase 1: Async Pipeline 구현

### 3.1 구현 전략 (iGPU 특화)

Thor는 통합 메모리이므로 "swap"이 아니라 **캐시 prefetch**:

```python
# scripts/inference/260524_async_pipeline.py

import torch
import ctypes

def prefetch_to_gpu_l2(tensor: torch.Tensor, stream: torch.cuda.Stream) -> None:
    """
    텐서를 GPU L2 캐시로 미리 올린다.
    iGPU(통합 메모리)에서 cudaMemPrefetchAsync 사용.
    """
    cuda_rt = ctypes.CDLL("libcudart.so")
    cuda_rt.cudaMemPrefetchAsync(
        ctypes.c_void_p(tensor.data_ptr()),
        ctypes.c_size_t(tensor.numel() * tensor.element_size()),
        ctypes.c_int(0),  # device 0
        ctypes.c_void_p(stream.cuda_stream)
    )


class AsyncPipelineDecoder:
    """
    RT-Swap + Demand Layering 아이디어를 Alpamayo decode에 적용.
    
    각 decode step:
      prefetch_stream: Layer(i+1) weights → GPU L2 prefetch
      compute_stream:  Layer(i) forward pass
    
    두 stream이 overlap → L2 hit rate 향상 → effective BW 증가
    """
    
    def __init__(self, model):
        self.model = model
        self.prefetch_stream = torch.cuda.Stream()
        self.compute_stream  = torch.cuda.Stream()
        self.layers = list(model.language_model.model.layers)  # 28 LM layers
    
    def decode_step(self, hidden_states, **kwargs):
        N = len(self.layers)
        
        # 첫 레이어 prefetch (파이프라인 프라이밍)
        prefetch_to_gpu_l2(self._get_layer_weights(0), self.prefetch_stream)
        torch.cuda.current_stream().wait_stream(self.prefetch_stream)
        
        for i in range(N):
            # (A) 다음 레이어 prefetch 시작 (비동기)
            if i + 1 < N:
                with torch.cuda.stream(self.prefetch_stream):
                    prefetch_to_gpu_l2(
                        self._get_layer_weights(i + 1),
                        self.prefetch_stream
                    )
            
            # (B) 현재 레이어 compute (동시에 실행됨!)
            with torch.cuda.stream(self.compute_stream):
                hidden_states = self.layers[i](hidden_states, **kwargs)
            
            # (C) 둘 다 끝날 때까지 대기
            torch.cuda.current_stream().wait_stream(self.compute_stream)
            torch.cuda.current_stream().wait_stream(self.prefetch_stream)
        
        return hidden_states
    
    def _get_layer_weights(self, layer_idx):
        """레이어의 주요 weight tensor 반환 (prefetch 대상)"""
        layer = self.layers[layer_idx]
        # MLP weights (가장 큰 것, 96 MB)
        return layer.mlp.gate_proj.weight
```

baseline_times = []

### 4.1 GPU L2 Persistent Residency (KV Cache 고정)

```python
# KV Cache를 GPU L2에 고정하여 attention 연산 가속
# 참고: CUDA 11.2+, SM 8.0+ 지원 (Thor SM 11.0 ✅)

import ctypes

CUDA_STREAM_ATTRIBUTE_ACCESS_POLICY_WINDOW = 1

class CudaAccessPolicyWindow(ctypes.Structure):
    _fields_ = [
        ("base_ptr", ctypes.c_void_p),
        ("num_bytes", ctypes.c_size_t),
        ("hitRatio", ctypes.c_float),
        ("hitProp", ctypes.c_int),    # cudaAccessPropertyPersisting = 1
        ("missProp", ctypes.c_int),   # cudaAccessPropertyStreaming = 2
    ]

def pin_kv_cache_to_l2(kv_cache_tensor, stream):
    """KV 캐시를 GPU L2에 고정 (evict 방지)"""
    cuda_rt = ctypes.CDLL("libcudart.so")
    
    # L2 persistent cache 크기 설정 (최대 32 MB 중 일부 예약)
    l2_persist_size = min(
        kv_cache_tensor.numel() * kv_cache_tensor.element_size(),
        16 * 1024 * 1024  # 16 MB (L2의 절반)
    )
    cuda_rt.cudaDeviceSetLimit(0x05, l2_persist_size)  # cudaLimitPersistingL2CacheSize
    
    # Access policy window 설정
    window = CudaAccessPolicyWindow()
    window.base_ptr  = kv_cache_tensor.data_ptr()
    window.num_bytes = kv_cache_tensor.numel() * kv_cache_tensor.element_size()
    window.hitRatio  = 1.0  # 100% L2 유지 시도
    window.hitProp   = 1    # cudaAccessPropertyPersisting
    window.missProp  = 1    # cudaAccessPropertyPersisting
    
    # Stream에 적용
    cuda_rt.cudaStreamSetAttribute(stream.cuda_stream, 1, ctypes.byref(window))
```

### 4.2 Weight Access Streaming Mode (L2 bypass)

```python
def set_weight_streaming(weight_tensor, stream):
    """
    Weight 접근 시 L2 bypass → L2를 KV cache 전용으로 확보
    """
    window = CudaAccessPolicyWindow()
    window.base_ptr  = weight_tensor.data_ptr()
    window.num_bytes = weight_tensor.numel() * weight_tensor.element_size()
    window.hitRatio  = 0.0  # L2에 저장 안 함
    window.hitProp   = 2    # cudaAccessPropertyStreaming  
    window.missProp  = 2    # cudaAccessPropertyStreaming
    
    cuda_rt.cudaStreamSetAttribute(stream.cuda_stream, 1, ctypes.byref(window))
```

### 5.1 성과 예측 (실험 전 추정)

| 최적화 기법 | 예상 speedup | 근거 |
|------------|-------------|------|
| Async Pipeline (cudaMemPrefetchAsync) | 1.1 ~ 1.5× | 현재 memory-bound, prefetch로 일부 중첩 |
| L2 Persistent KV Cache | 1.05 ~ 1.2× | KV cache(7-28MB)가 L2 히트 시 5× 빠름 |
| 복합 적용 | 1.2 ~ 1.8× | 두 기법 상호보완 |
| Multi-DNN 스케줄링 | 1.5 ~ 3.0× | RT-Swap 논문 결과 참고 |
